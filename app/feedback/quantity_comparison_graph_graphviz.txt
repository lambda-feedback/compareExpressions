    START [shape="polygon" style="filled" fillcolor="#FFFFCC"]

    PREFIX_IS_LARGE [label="unit(response) >= 1000*unit(answer)" tooltip="Check if prefix is much larger for the response than the answer" shape="polygon" style="filled" fillcolor="#FFFFCC"]
    PREFIX_IS_SMALL [label="unit(response)*1000 <= unit(answer)" tooltip="Check if prefix is much smaller for the response than the answer" shape="polygon" style="filled" fillcolor="#FFFFCC"]
    START -> DIMENSION_MATCH
        DIMENSION_MATCH -> QUANTITY_MATCH [label="True"]
            QUANTITY_MATCH -> PREFIX_IS_LARGE [label="True"]
                PREFIX_IS_LARGE -> PREFIX_IS_SMALL [label="True"]
                    PREFIX_IS_SMALL -> END [label="True"]
                    PREFIX_IS_SMALL -> END [label="False"]
                PREFIX_IS_LARGE -> END [label="False"]
